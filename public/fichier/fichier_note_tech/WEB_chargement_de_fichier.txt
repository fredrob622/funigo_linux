  <script>
    async function openTxtInNewTab(fichierAOuvrir) {
      try {
        // Correction : utilisation des littéraux de gabarit pour la variable
        const response = await fetch(`../../public/fichier/fichier_note_tech/${fichierAOuvrir}`);
        if (!response.ok) throw new Error("Fichier non trouvé ou erreur de réseau");
        
        const text = await response.text();
        const newTab = window.open();
        newTab.document.write(`<pre>${text}</pre>`); // '<pre>' pour conserver la mise en forme
      } catch (error) {
        console.error("Erreur :", error);
        alert("Impossible d'ouvrir le fichier. Vérifiez le nom ou le chemin.");
      }
    }  
  </script>

  <button onclick="openTxtInNewTab('WEB_chargement_de_fichier.txt')">Ouvrir fichier TXT</button>
</body>
